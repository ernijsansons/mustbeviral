import { NavBar } from './components/NavBar';
import { ProtectedRoute } from './components/ProtectedRoute';
import { CriticalErrorBoundary, PageErrorBoundary } from './components/ErrorBoundary';
import { DevDebugger } from './components/DevDebugger';
import { env, validateEnvironment } from './lib/env';
import { QueryProvider, QueryErrorBoundary } from './providers/QueryProvider';
import { ThemeProvider } from './providers/ThemeProvider';
import { logger } from './lib/logging/productionLogger';
import { errorTracker } from './lib/monitoring/errorTracking';
import { performanceMonitor } from './lib/monitoring/performanceMonitor';
 
function App() {
  // Initialize monitoring systems
  useEffect(() => {
    // Development mode logging
    if (env.APP_ENV === 'development') {
      console.log('🚀 App initialization started');
      console.log('📊 Environment:', env);
    }
 
    // Initialize error tracking
    try {
      errorTracker.initialize({
        environment: env.APP_ENV,
        release: env.APP_VERSION,
        enablePerformanceMonitoring: true
      });
 
      if (env.APP_ENV === 'development') {
        console.log('✅ Error tracking initialized');
      }
    } catch (err) {
      console.error('❌ Failed to initialize error tracking:', err);
    }
 
    // Start performance monitoring (safely)
    if (performanceMonitor && performanceMonitor.startMonitoring) {
      performanceMonitor.startMonitoring();
    }
 
    // Add global error tracking
    const originalError = window.onerror;
    window.onerror = (message, source, lineno, colno, error) => {
      if (error) {
        errorTracker.captureError(error);
      }
      return originalError ? originalError(message, source, lineno, colno, error) : false;
    };
 
    return () => {
      if (performanceMonitor && performanceMonitor.stopMonitoring) {
        performanceMonitor.stopMonitoring();
      }
    };
  }, []);
 
  // Validate environment configuration on app start
  const { isValid, errors } = validateEnvironment();
 
  if (!isValid) {
    if (env.APP_ENV === 'production') {
      logger.error('Environment validation failed', undefined, {
        component: 'App',
        action: 'environment_validation',
        metadata: { errors }
      });
    } else {
      console.warn('⚠️ Environment validation warnings:', errors);
    }
  } else if (env.APP_ENV === 'development') {
    console.log('✅ Environment validation passed');
  }
 
  return (
    <CriticalErrorBoundary>
      <QueryErrorBoundary>
        <QueryProvider>
          <AuthProvider>
            <ThemeProvider>
              <Router>
              <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100">
                <NavBar />
 
                <main className="min-h-screen">
                  <Suspense fallback={
                    <div className="flex items-center justify-center min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100">
                      <div className="text-center">
                        <div className="animate-spin rounded-full h-16 w-16 border-b-2 border-indigo-600 mx-auto mb-4"></div>
                        <h3 className="text-lg font-medium text-gray-900 mb-2">Loading...</h3>
                        <p className="text-gray-600">Please wait while we prepare your content</p>
                      </div>
                    </div>
                  }>
                    <Switch>
                      <Route path="/">
                        <PageErrorBoundary context="HomePage">
                          <HomePage />
                        </PageErrorBoundary>
                      </Route>
 
                      <Route path="/login">
                        <PageErrorBoundary context="LoginPage">
                          <LoginPage />
                        </PageErrorBoundary>
                      </Route>
 
                      <Route path="/onboard">
                        <PageErrorBoundary context="OnboardPage">
                          <OnboardPage />
                        </PageErrorBoundary>
                      </Route>
 
                      <Route path="/demo">
                        <PageErrorBoundary context="DesignSystemDemo">
                          <DesignSystemDemo />
                        </PageErrorBoundary>
                      </Route>
 
                      {/* Protected Routes */}
                      <Route path="/dashboard">
                        <ProtectedRoute>
                          <PageErrorBoundary context="Dashboard">
                            <Dashboard />
                          </PageErrorBoundary>
                        </ProtectedRoute>
                      </Route>
 
                      <Route path="/content">
                        <ProtectedRoute>
                          <PageErrorBoundary context="ContentPage">
                            <ContentPage />
                          </PageErrorBoundary>
                        </ProtectedRoute>
                      </Route>
 
                      <Route path="/matches">
                        <ProtectedRoute>
                          <PageErrorBoundary context="MatchesPage">
                            <MatchesPage />
                          </PageErrorBoundary>
                        </ProtectedRoute>
                      </Route>
 
                      <Route path="/monitoring">
                        <ProtectedRoute>
                          <PageErrorBoundary context="MonitoringDashboard">
                            <MonitoringDashboard />
                          </PageErrorBoundary>
                        </ProtectedRoute>
                      </Route>
 
                    {/* Catch-all route for 404 */}
                    <Route path="/:rest*">
                      <PageErrorBoundary context="404Page">
                        <div className="max-w-7xl mx-auto py-6 sm:px-6 lg:px-8">
                          <div className="px-4 py-6 sm:px-0">
                            <div className="bg-white rounded-lg shadow p-6 text-center">
                              <h2 className="text-2xl font-bold text-gray-900 mb-4">Page Not Found</h2>
                              <p className="text-gray-600 mb-4">The page you're looking for doesn't exist.</p>
                              <a
                                href="/"
                                className="bg-indigo-600 text-white px-4 py-2 rounded-md font-medium hover:bg-indigo-700 focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:ring-offset-2"
                              >
                                Go Home
                              </a>
                            </div>
                          </div>
                        </div>
                      </PageErrorBoundary>
                    </Route>
                  </Switch>
                  </Suspense>
                </main>
                </div>
              </Router>
            </ThemeProvider>
          </AuthProvider>
        </QueryProvider>
      </QueryErrorBoundary>
 
      {/* Development Debug Panel */}
      <DevDebugger />
    </CriticalErrorBoundary>
  );
}
 
export default App;
